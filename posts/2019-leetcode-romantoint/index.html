var romanList = [
    {name: 'M', value: 1000},
    {name: 'D', value: 500},
    {name: 'C', value: 100},
    {name: 'L', value: 50},
    {name: 'X', value: 10},
    {name: 'V', value: 5},
    {name: 'I', value: 1}
]
var getRomanMap = (s) => {
    const mapData = new Map();
    romanList.forEach(v => {
        mapData.set(v.name, v.value);
    });
    console.log(mapData);
    return mapData;
}
var romanToInt = function(s) {
    const romanMap = getRomanMap(s);
    const strArr = s.split('');
    let result = 0;
    let nowData = 0;
    let next = 0;
    for(let i = 0; i < strArr.length; i++) {
        nowData = romanMap.get(strArr[i]);
        next = romanMap.get(strArr[i + 1]);
        if (nowData < next) {
            result -= nowData;
        } else {
            result += nowData;
        }
    }
    console.log(result);
    return result;
};
